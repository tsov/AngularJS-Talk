// todo_controller.js
$scope.addTodo = function() {
  $scope.items.push({name: $scope.newTodo, done: false});
  $scope.newTodo = '';
}

// todo_controllerSpec.js
describe('#addTodo', function() {
  beforeEach(function() {
    scope.newTodo = 'new todo';
    scope.addTodo();
  })

  it('should add a todo', function() {
    expect(scope.items).toContain({name:'new todo', done:false});
  })

  it('should empty newTodo variable', function() {
    expect(scope.newTodo).toEqual('');
  })
});
